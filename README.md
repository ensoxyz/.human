CURRENCY TEXT, 
GDP_T1 INTEGER,
GDP_T2 INTEGER, 
PPP_T1 INTEGER,
PPP_T2 INTEGER, 
MEDIAN_I INTEGER, 
MEDIAN_H INTEGER, 
POPULATION INTEGER, 
NL INTEGER, 
EN INTEGER, 
"0-18" INTEGER, 
"19-25" INTEGER, 
"26-34" INTEGER, 
"35-54" INTEGER, 
"55-64" INTEGER, 
"65+" INTEGER, 
AFR INTEGER, 
ARB INTEGER, 
ASN INTEGER, 
LTN INTEGER, 
IND INTEGER, 
NTV INTEGER, 
WHT INTEGER,
